## 5.1 暗色主题系统

### 设计思路

本项目采用 **GitHub Dark + VS Code** 风格的暗色主题，技术方案：

1. **CSS 变量覆盖** — 重写 Arco Design 的 CSS 变量
2. **`arco-theme="dark"`** — 触发 Arco 内置暗色模式
3. **Pinia 状态管理** — 全局主题状态
4. **localStorage 持久化** — 记住用户选择
5. **系统偏好检测** — 自动跟随系统主题

### Pinia Theme Store

```typescript
// src/store/theme.ts
import { defineStore } from "pinia";
import { ref } from "vue";

export const useThemeStore = defineStore("theme", () => {
  // 初始化逻辑：
  // 1. 优先读取 localStorage
  // 2. 其次检测系统偏好
  // 3. 默认亮色
  const isDark = ref<boolean>(
    localStorage.getItem("theme") === "dark" ||
      (!localStorage.getItem("theme") &&
        window.matchMedia("(prefers-color-scheme: dark)").matches)
  );

  const toggleDark = () => {
    isDark.value = !isDark.value;
    applyTheme();
  };

  const setDark = (value: boolean) => {
    isDark.value = value;
    applyTheme();
  };

  const applyTheme = () => {
    // 1. 设置 body 的 arco-theme 属性
    if (isDark.value) {
      document.body.setAttribute("arco-theme", "dark");
    } else {
      document.body.removeAttribute("arco-theme");
    }
    // 2. 持久化到 localStorage
    localStorage.setItem("theme", isDark.value ? "dark" : "light");
  };

  // 初始化时立即应用
  applyTheme();

  // 监听系统主题变化（仅在用户未手动选择时生效）
  window.matchMedia("(prefers-color-scheme: dark)").addEventListener("change", (e) => {
    if (!localStorage.getItem("theme")) {
      isDark.value = e.matches;
      applyTheme();
    }
  });

  return { isDark, toggleDark, setDark };
});
```

## 5.2 AI 聊天助手

### 功能概述

- 基于 **Ollama** 的本地 AI 对话
- **SSE（Server-Sent Events）** 流式输出，打字机效果
- 聊天记录持久化到 `ai_message` 表
- AI 回复支持 Markdown、代码、LaTeX 公式

### 前端实现（AIAssistantView）

```vue
<!-- src/views/AIAssistantView.vue 简化 -->
<template>
  <div class="chat-container">
    <!-- 消息列表 -->
    <div class="messages" ref="messagesRef">
      <div v-for="(msg, i) in messages" :key="i" :class="['message', msg.role]">
        <!-- 用户消息：纯文本 -->
        <div v-if="msg.role === 'user'" class="bubble">{{ msg.content }}</div>
        <!-- AI 消息：Markdown 渲染 -->
        <MdPreview v-else :value="msg.content" />
      </div>
      <!-- 打字机动画 -->
      <div v-if="loading" class="typing-indicator">AI 正在思考...</div>
    </div>

    <!-- 输入区 -->
    <div class="input-area">
      <a-textarea v-model="input" placeholder="输入问题..." @pressEnter="sendMessage" />
      <a-button type="primary" :loading="loading" @click="sendMessage">发送</a-button>
    </div>
  </div>
</template>

<script setup lang="ts">
import { ref, nextTick, onMounted } from "vue";
import axios from "axios";
import MdPreview from "@/components/MdPreview.vue";

const messages = ref<Array<{ role: string; content: string }>>([]);
const input = ref("");
const loading = ref(false);

// 加载历史消息
onMounted(async () => {
  const res = await axios.get("/api/ai/messages");
  if (res.data?.code === 0) {
    messages.value = res.data.data || [];
  }
});

const sendMessage = async () => {
  if (!input.value.trim() || loading.value) return;

  // 1. 添加用户消息
  const userMsg = input.value;
  messages.value.push({ role: "user", content: userMsg });
  input.value = "";

  // 2. 添加 AI 空消息（用于流式追加）
  messages.value.push({ role: "assistant", content: "" });
  loading.value = true;

  // 3. 建立 SSE 连接
  const eventSource = new EventSource(
    `/api/ai/stream?prompt=${encodeURIComponent(userMsg)}`
  );

  // 4. 接收流式数据
  eventSource.onmessage = (event) => {
    const data = event.data;
    if (data === "[DONE]") {
      // 流结束
      eventSource.close();
      loading.value = false;
      return;
    }
    // 追加到 AI 消息（打字机效果）
    const lastMsg = messages.value[messages.value.length - 1];
    lastMsg.content += data;
    // 自动滚动到底部
    nextTick(() => {
      scrollToBottom();
    });
  };

  eventSource.onerror = () => {
    eventSource.close();
    loading.value = false;
  };
};

const scrollToBottom = () => {
  const container = messagesRef.value;
  if (container) {
    container.scrollTop = container.scrollHeight;
  }
};
</script>
```

### 后端接口

```java
// AiChatController.java
@RestController
@RequestMapping("/ai")
public class AiChatController {

    @Value("${ollama.base-url}")
    private String ollamaBaseUrl;

    @Value("${ollama.model}")
    private String ollamaModel;

    // SSE 流式输出
    @GetMapping("/stream")
    public SseEmitter streamChat(@RequestParam String prompt,
                                  HttpServletRequest request) {
        SseEmitter emitter = new SseEmitter(0L); // 无超时

        // 异步执行
        CompletableFuture.runAsync(() -> {
            try {
                // 调用 Ollama /api/generate
                URL url = new URL(ollamaBaseUrl + "/api/generate");
                HttpURLConnection conn = (HttpURLConnection) url.openConnection();
                conn.setRequestMethod("POST");
                conn.setDoOutput(true);
                conn.setRequestProperty("Content-Type", "application/json");

                // 构建请求体
                String body = String.format(
                    "{\"model\":\"%s\",\"prompt\":\"%s\",\"stream\":true}",
                    ollamaModel, escapeJson(prompt)
                );
                conn.getOutputStream().write(body.getBytes());

                // 流式读取响应
                BufferedReader reader = new BufferedReader(
                    new InputStreamReader(conn.getInputStream())
                );
                String line;
                while ((line = reader.readLine()) != null) {
                    if (line.isEmpty()) continue;
                    // 解析 JSON，提取 response 字段
                    JSONObject json = JSON.parseObject(line);
                    String token = json.getString("response");
                    if (token != null) {
                        emitter.send(token);  // 推送给前端
                    }
                    if (json.getBooleanValue("done")) {
                        emitter.send("[DONE]");
                        break;
                    }
                }
                emitter.complete();
            } catch (Exception e) {
                emitter.completeWithError(e);
            }
        });

        return emitter;
    }

    // 获取历史消息
    @GetMapping("/messages")
    public BaseResponse<List<AiMessage>> getMessages(HttpServletRequest request) {
        Long userId = JwtInterceptor.getUserId(request);
        return ResultUtils.success(aiMessageService.getMessages(userId));
    }
}
```

<template>
  <div class="stats-page">
    <!-- ECharts 饼图 -->
    <div ref="chartRef" style="width: 100%; height: 400px"></div>

<script setup lang="ts">
import { ref, onMounted } from "vue";
import * as echarts from "echarts";
import axios from "axios";

const chartRef = ref<HTMLElement>();

onMounted(async () => {
  const res = await axios.get("/api/statistics/language-distribution");
  const data = res.data.data;

  // 初始化 ECharts
  const chart = echarts.init(chartRef.value!);

  // 配置饼图
  chart.setOption({
    tooltip: { trigger: "item", formatter: "{b}: {c} ({d}%)" },
    legend: { orient: "vertical", left: "left" },
    series: [{
      type: "pie",
      radius: "50%",
      data: data.map((item: any) => ({
        name: item.language,
        value: item.count,
      })),
      emphasis: {
        itemStyle: { shadowBlur: 10, shadowColor: "rgba(0, 0, 0, 0.5)" },
      },
    }],
  });
});
