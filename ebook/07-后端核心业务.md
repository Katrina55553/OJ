## 7.1 用户系统

### 注册流程

```java
// UserServiceImpl.java
@Override
public long userRegister(UserRegisterRequest request) {
    String account = request.getUserAccount();
    String password = request.getUserPassword();
    String confirmPassword = request.getCheckPassword();

    // 1. 参数校验
    if (StringUtils.isAnyBlank(account, password, confirmPassword)) {
        throw new BusinessException(ErrorCode.PARAMS_ERROR, "参数为空");
    }
    if (account.length() < 4) {
        throw new BusinessException(ErrorCode.PARAMS_ERROR, "账号长度不能小于 4");
    }
    if (password.length() < 8) {
        throw new BusinessException(ErrorCode.PARAMS_ERROR, "密码长度不能小于 8");
    }
    if (!password.equals(confirmPassword)) {
        throw new BusinessException(ErrorCode.PARAMS_ERROR, "两次密码不一致");
    }

    // 2. 账号查重
    QueryWrapper<User> queryWrapper = new QueryWrapper<>();
    queryWrapper.eq("userAccount", account);
    long count = userMapper.selectCount(queryWrapper);
    if (count > 0) {
        throw new BusinessException(ErrorCode.PARAMS_ERROR, "账号已存在");
    }

    // 3. 密码加密（MD5）
    String encryptPassword = DigestUtils.md5DigestAsHex(password.getBytes());

    // 4. 入库
    User user = new User();
    user.setUserAccount(account);
    user.setUserPassword(encryptPassword);
    user.setUserRole(UserRoleEnum.USER.getValue());
    boolean save = this.save(user);
    if (!save) {
        throw new BusinessException(ErrorCode.SYSTEM_ERROR, "注册失败");
    }
    return user.getId();
}
```

### 登录流程

```java
@Override
public LoginUserVO userLogin(UserLoginRequest request, HttpServletRequest httpRequest) {
    String account = request.getUserAccount();
    String password = request.getUserPassword();

    // 1. 查询用户
    QueryWrapper<User> queryWrapper = new QueryWrapper<>();
    queryWrapper.eq("userAccount", account);
    User user = userMapper.selectOne(queryWrapper);
    if (user == null) {
        throw new BusinessException(ErrorCode.PARAMS_ERROR, "账号不存在");
    }

    // 2. 校验密码
    String encryptPassword = DigestUtils.md5DigestAsHex(password.getBytes());
    if (!encryptPassword.equals(user.getUserPassword())) {
        throw new BusinessException(ErrorCode.PARAMS_ERROR, "密码错误");
    }

    // 3. 检查是否被封禁
    if (UserRoleEnum.BAN.getValue().equals(user.getUserRole())) {
        throw new BusinessException(ErrorCode.NO_AUTH_ERROR, "账号已被封禁");
    }

    // 4. 生成 JWT
    String token = JwtUtils.createToken(user.getId());

    // 5. 返回脱敏用户信息
    return getSafetyUser(user, token);
}
```

private LoginUserVO getSafetyUser(User user, String token) {
    LoginUserVO loginUserVO = new LoginUserVO();
    loginUserVO.setId(user.getId());

    loginUserVO.setUserName(user.getUserName());

@Override
public Page<Question> listQuestionByPage(QuestionQueryRequest request) {
    // 1. 构建分页对象
    Page<Question> page = new Page<>(request.getCurrent(), request.getPageSize());

    // 2. 构建查询条件
    QueryWrapper<Question> queryWrapper = new QueryWrapper<>();
    queryWrapper.like(StringUtils.isNotBlank(request.getTitle()),
                      "title", request.getTitle());
    queryWrapper.eq(StringUtils.isNotBlank(request.getDifficulty()),
                    "difficulty", request.getDifficulty());
    queryWrapper.like(StringUtils.isNotBlank(request.getTags()),
                      "tags", request.getTags());
    queryWrapper.orderByDesc("createTime");

    // 3. 执行分页查询
    return questionMapper.selectPage(page, queryWrapper);
}

### 点赞（PostThumbController）

```java
@PostMapping("/thumb")
public BaseResponse<Boolean> doPostThumb(@RequestBody PostThumbRequest request,
                                          HttpServletRequest httpRequest) {
    Long postId = request.getPostId();
    Long userId = JwtInterceptor.getUserId(httpRequest);

    // 幂等设计：已点赞则取消，未点赞则点赞
    PostThumb postThumb = new PostThumb();
    postThumb.setPostId(postId);
    postThumb.setUserId(userId);

    // 查询是否已点赞
    QueryWrapper<PostThumb> queryWrapper = new QueryWrapper<>();
    queryWrapper.eq("postId", postId).eq("userId", userId);
    PostThumb existing = postThumbMapper.selectOne(queryWrapper);

    if (existing != null) {
        // 已点赞 → 取消（硬删除）
        postThumbMapper.delete(queryWrapper);
        return ResultUtils.success(false); // 返回 false 表示取消
    } else {
        // 未点赞 → 点赞
        postThumbMapper.insert(postThumb);
        return ResultUtils.success(true);  // 返回 true 表示点赞
    }
}
```

## 7.4 AI 聊天后端

### SSE 流式输出

```java
@GetMapping("/stream")
public SseEmitter streamChat(@RequestParam String prompt,
                              HttpServletRequest request) {
    Long userId = JwtInterceptor.getUserId(request);
    SseEmitter emitter = new SseEmitter(0L); // 无超时

    CompletableFuture.runAsync(() -> {
        try {
            // 1. 保存用户消息
            saveMessage(userId, "user", prompt);

            // 2. 调用 Ollama
            URL url = new URL(ollamaBaseUrl + "/api/generate");
            HttpURLConnection conn = (HttpURLConnection) url.openConnection();
            conn.setRequestMethod("POST");
            conn.setDoOutput(true);
            conn.setRequestProperty("Content-Type", "application/json");

            String body = String.format(
                "{\"model\":\"%s\",\"prompt\":\"%s\",\"stream\":true}",
                ollamaModel, escapeJson(prompt)
            );
            conn.getOutputStream().write(body.getBytes());

            // 3. 流式读取并推送
            BufferedReader reader = new BufferedReader(
                new InputStreamReader(conn.getInputStream())
            );
            StringBuilder fullResponse = new StringBuilder();
            String line;
            while ((line = reader.readLine()) != null) {
                if (line.isEmpty()) continue;
                JSONObject json = JSON.parseObject(line);
                String token = json.getString("response");
                if (token != null) {
                    fullResponse.append(token);
                    emitter.send(token);
                }
                if (json.getBooleanValue("done")) {
                    emitter.send("[DONE]");
                    break;
                }
            }

            // 4. 保存 AI 完整回复
            saveMessage(userId, "assistant", fullResponse.toString());
            emitter.complete();
        } catch (Exception e) {
            emitter.completeWithError(e);
        }
    });

    return emitter;
}
```

## 7.6 文件上传

```java
@RestController
@RequestMapping("/file")
public class FileController {

    @Autowired
    private CosManager cosManager;

    @PostMapping("/upload")
    public BaseResponse<String> uploadFile(@RequestParam("file") MultipartFile file) {
        // 1. 校验文件类型
        String contentType = file.getContentType();
        if (!ALLOWED_TYPES.contains(contentType)) {
            return ResultUtils.error(ErrorCode.PARAMS_ERROR, "不支持的文件类型");
        }

        // 2. 校验文件大小（1MB）
        if (file.getSize() > 1024 * 1024) {
            return ResultUtils.error(ErrorCode.PARAMS_ERROR, "文件大小不能超过 1MB");
        }

        // 3. 生成唯一文件名
        String filename = "avatar/" + UUID.randomUUID() + getExtension(file);

        // 4. 上传到 COS
        cosManager.uploadFile(filename, file.getInputStream());

        // 5. 返回文件访问 URL
        return ResultUtils.success(cosManager.getFileUrl(filename));
    }
}
```

## 7.7 小结

| 要点 | 内容 |
|------|------|
| **用户系统** | 注册（参数校验+查重+MD5加密）、登录（JWT）、脱敏（VO）|
| **题目管理** | CRUD + 分页查询 + JSON 字段 + VO 脱敏（不返回 answer）|
| **帖子系统** | CRUD + 点赞（幂等/硬删除）+ 收藏 + ES 搜索 |
| **AI 聊天** | SSE 流式输出 + Ollama 集成 + 消息持久化 |
| **代码执行** | 即时执行（同步）vs 判题提交（异步）|
| **文件上传** | COS 集成 + 类型校验 + 大小限制 |
