## 4.3 Monaco Editor 集成

### CodeEditor 组件

Monaco Editor 是 VS Code 的核心编辑器组件，提供语法高亮、自动补全、代码折叠等功能。

```vue
<!-- src/components/CodeEditor.vue -->
<template>
  <div ref="editorContainer" :style="{ height: '100%' }"></div>
</template>

<script setup lang="ts">
import { ref, onMounted, watch, onUnmounted } from "vue";
import * as monaco from "monaco-editor";

// Props
const props = withDefaults(defineProps<{
  modelValue?: string;      // v-model 绑定
  language?: string;        // 编程语言
  readOnly?: boolean;       // 只读模式
  theme?: string;           // 主题（vs-dark / vs-light）
  minimapEnabled?: boolean; // 小地图
}>(), {
  modelValue: "",
  language: "cpp",
  readOnly: false,
  theme: "vs-dark",
  minimapEnabled: true,
});

const emit = defineEmits(["update:modelValue"]);
const editorContainer = ref<HTMLElement>();
let editor: monaco.editor.IStandaloneCodeEditor | null = null;

onMounted(() => {
  // 创建编辑器实例
  editor = monaco.editor.create(editorContainer.value!, {
    value: props.modelValue,
    language: props.language,
    theme: props.theme,
    readOnly: props.readOnly,
    minimap: { enabled: props.minimapEnabled },
    automaticLayout: true,       // 自动适应容器大小
    smoothScrolling: true,       // 平滑滚动
    folding: true,               // 代码折叠
    wordWrap: "on",              // 自动换行
    suggestOnTriggerCharacters: true,  // 触发字符时显示建议
  });

  // 监听内容变化，同步到 v-model
  editor.onDidChangeModelContent(() => {
    emit("update:modelValue", editor!.getValue());
  });
});

// 响应式：语言变化
watch(() => props.language, (newLang) => {
  if (editor) {
    monaco.editor.setModelLanguage(editor.getModel()!, newLang);
  }
});

// 响应式：主题变化
watch(() => props.theme, (newTheme) => {
  if (editor) {
    monaco.editor.setTheme(newTheme);
  }
});

// 暴露方法供父组件调用
const fillValue = (val: string) => {
  if (editor) {
    editor.setValue(val);
  }
};

defineExpose({ fillValue });

onUnmounted(() => {
  editor?.dispose();
});
</script>
```

### LanguageCodeEditor 组件

在 CodeEditor 基础上增加语言选择和默认模板：

```vue
<!-- src/components/LanguageCodeEditor.vue -->
<template>
  <div>
    <!-- 语言选择下拉框 -->
    <a-select v-model="currentLanguage" @change="handleLanguageChange">
      <a-option value="cpp">C++</a-option>
      <a-option value="java">Java</a-option>
      <a-option value="python">Python</a-option>
      <a-option value="go">Go</a-option>
      <a-option value="javascript">JavaScript</a-option>
    </a-select>
    <!-- 代码编辑器 -->
    <CodeEditor
      v-model="currentCode"
      :language="currentLanguage"
      theme="vs-dark"
    />
  </div>
</template>

<script setup lang="ts">
import { ref, watch } from "vue";
import CodeEditor from "./CodeEditor.vue";

// 每种语言的默认代码模板
const codeTemplates: Record<string, string> = {
  cpp: `#include <iostream>
using namespace std;

int main() {
    // 在此编写代码
    return 0;
}`,
  java: `import java.util.*;

public class Main {
    public static void main(String[] args) {
        // 在此编写代码
    }
}`,
  python: `# 在此编写代码
def main():
    pass

if __name__ == "__main__":
    main()`,
  go: `package main

import "fmt"

func main() {
    // 在此编写代码
}`,
  javascript: `// 在此编写代码
function main() {
}

main();`,
};

const currentLanguage = ref("cpp");
const currentCode = ref(codeTemplates.cpp);

// 切换语言时自动填充模板（仅当代码为空或匹配旧模板时）
const handleLanguageChange = (newLang: string) => {
  const oldTemplate = codeTemplates[currentLanguage.value];
  if (!currentCode.value || currentCode.value === oldTemplate) {
    currentCode.value = codeTemplates[newLang];
  }
};
</script>
```

## 4.4 Markdown 生态

### 编辑器（MdEditor）

基于 **ByteMD** 库，提供富文本 Markdown 编辑体验：

```vue
<!-- src/components/MdEditor.vue -->
<template>
  <Editor
    :value="modelValue"
    :plugins="plugins"
    @change="handleChange"
  />
</template>

<script setup lang="ts">
import { Editor } from "@bytemd/vue-next";
import gfm from "@bytemd/plugin-gfm";
import highlight from "@bytemd/plugin-highlight";

const plugins = [
  gfm(),       // GitHub Flavored Markdown（表格、任务列表等）
  highlight(), // 代码语法高亮
];

const props = defineProps<{ modelValue: string }>();
const emit = defineEmits(["update:modelValue"]);

const handleChange = (v: string) => {
  emit("update:modelValue", v);
};
</script>
```

### 预览器（MdPreview）

基于 **markdown-it** 库，支持代码高亮和数学公式：

```vue
<!-- src/components/MdPreview.vue -->
<template>
  <div class="markdown-body" v-html="htmlContent"></div>
</template>

<script setup lang="ts">
import { computed } from "vue";
import MarkdownIt from "markdown-it";
import highlight from "markdown-it-highlightjs";
import texmath from "markdown-it-texmath";
import katex from "katex";

const props = defineProps<{ value: string }>();

// 配置 markdown-it
const md = new MarkdownIt({
  html: false,  // 禁止原始 HTML（XSS 防护）
  linkify: true,
  typographer: true,
})
  .use(highlight)           // 代码语法高亮
  .use(texmath, { katex }); // LaTeX 数学公式

// 渲染 Markdown 为 HTML
const htmlContent = computed(() => {
  return md.render(props.value || "");
});
</script>
```

### 两个 Markdown 管线

| 用途 | 组件 | 库 | 特点 |
|------|------|-----|------|
| **编辑** | MdEditor | ByteMD | 工具栏、实时预览、GFM 支持 |
| **预览** | MdPreview | markdown-it | 代码高亮、LaTeX 公式、XSS 防护 |
