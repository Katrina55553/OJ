## 4.3 CodeMirror 6 集成

### CodeEditor 组件

CodeMirror 6 是一个轻量级、模块化的代码编辑器，相比 Monaco Editor 体积小 5 倍（~1MB vs 5MB+），构建时间从 50s 降到 6s，特别适合资源受限的部署环境（如 2GB VPS）。

```vue
<!-- src/components/CodeEditor.vue -->
<template>
  <div ref="editorRef" class="cm-editor-container" />
</template>

<script setup lang="ts">
import { ref, onMounted, watch, onBeforeUnmount, shallowRef } from "vue";
import { EditorState, Compartment } from "@codemirror/state";
import { EditorView, keymap, lineNumbers, highlightActiveLine } from "@codemirror/view";
import { defaultKeymap, history, historyKeymap } from "@codemirror/commands";
import { searchKeymap } from "@codemirror/search";
import { indentOnInput, bracketMatching, foldGutter, foldKeymap } from "@codemirror/language";
import { autocompletion, completionKeymap, closeBrackets, closeBracketsKeymap } from "@codemirror/autocomplete";
import { oneDark } from "@codemirror/theme-one-dark";
import { cpp } from "@codemirror/lang-cpp";
import { java } from "@codemirror/lang-java";
import { python } from "@codemirror/lang-python";
import { go } from "@codemirror/lang-go";
import { javascript } from "@codemirror/lang-javascript";

// Props（与原 Monaco 版本保持一致，方便替换）
const props = withDefaults(defineProps<{
  modelValue?: string;
  language?: string;
  readOnly?: boolean;
  theme?: "dark" | "light";
  minimapEnabled?: boolean; // 兼容旧 props，CodeMirror 6 无 minimap
}>(), {
  modelValue: "",
  language: "cpp",
  readOnly: false,
  theme: "dark",
  minimapEnabled: true,
});

const emit = defineEmits(["update:modelValue"]);
const editorRef = ref<HTMLDivElement | null>(null);
const editorView = shallowRef<EditorView | null>(null);

// Compartment 用于动态切换语言/主题，无需重建编辑器
const languageCompartment = new Compartment();
const themeCompartment = new Compartment();
const readOnlyCompartment = new Compartment();

function getLanguageExtension(lang: string) {
  switch (lang.toLowerCase()) {
    case "cpp": return cpp();
    case "java": return java();
    case "python": return python();
    case "go": return go();
    case "javascript": return javascript();
    default: return cpp();
  }
}

onMounted(() => {
  if (!editorRef.value) return;

  const updateListener = EditorView.updateListener.of((vu) => {
    if (vu.docChanged) {
      emit("update:modelValue", vu.state.doc.toString());
    }
  });

  const state = EditorState.create({
    doc: props.modelValue,
    extensions: [
      lineNumbers(),
      highlightActiveLine(),
      history(),
      foldGutter(),
      indentOnInput(),
      bracketMatching(),
      closeBrackets(),
      autocompletion(),
      keymap.of([
        ...closeBracketsKeymap,
        ...defaultKeymap,
        ...searchKeymap,
        ...historyKeymap,
        ...foldKeymap,
        ...completionKeymap,
      ]),
      languageCompartment.of(getLanguageExtension(props.language)),
      themeCompartment.of(props.theme === "dark" ? oneDark : []),
      readOnlyCompartment.of(EditorState.readOnly.of(props.readOnly)),
      EditorView.lineWrapping,
      updateListener,
    ],
  });

  editorView.value = new EditorView({ state, parent: editorRef.value });
});

// 响应式：语言变化（通过 compartment 动态切换，不重建编辑器）
watch(() => props.language, (newLang) => {
  editorView.value?.dispatch({
    effects: languageCompartment.reconfigure(getLanguageExtension(newLang)),
  });
});

// 响应式：主题变化
watch(() => props.theme, (newTheme) => {
  editorView.value?.dispatch({
    effects: themeCompartment.reconfigure(newTheme === "dark" ? oneDark : []),
  });
});

// 响应式：readOnly 变化
watch(() => props.readOnly, (newReadOnly) => {
  editorView.value?.dispatch({
    effects: readOnlyCompartment.reconfigure(EditorState.readOnly.of(newReadOnly)),
  });
});

// 暴露方法供父组件调用
const fillValue = (val: string) => {
  const view = editorView.value;
  if (!view) return;
  view.dispatch({ changes: { from: 0, to: view.state.doc.length, insert: val } });
};

defineExpose({ fillValue });

onBeforeUnmount(() => {
  editorView.value?.destroy();
});
</script>
```

### 为什么选 CodeMirror 6 而不是 Monaco

| 对比项 | Monaco Editor | CodeMirror 6 |
|--------|---------------|--------------|
| 依赖体积 | 5MB+ | ~1MB |
| Vite 构建时间 | 50s+ | 6s |
| 构建内存峰值 | 2GB+（2GB VPS OOM） | <500MB |
| 模块化 | 整体引入 | 按需引入语言包 |
| Worker | 需要注册 ESM worker | 无需 worker |

> **迁移背景**：项目原本用 Monaco Editor，但部署到 2GB VPS 后，Vite 构建时内存峰值超过 2GB，即使加 swap 也无法完成 build。换成 CodeMirror 6 后构建时间从 50s 降到 6s，内存峰值降到 500MB 以内。

### LanguageCodeEditor 完整实现

```vue
<!-- src/components/LanguageCodeEditor.vue -->
<template>
  <div>
    <!-- 语言选择下拉框 -->
    <a-select v-model="currentLanguage" @change="handleLanguageChange">
      <a-option value="cpp">C++</a-option>
      <a-option value="java">Java</a-option>
      <a-option value="python">Python</a-option>
      <a-option value="go">Go</a-option>
      <a-option value="javascript">JavaScript</a-option>
    </a-select>
    <!-- 代码编辑器 -->
    <CodeEditor
      v-model="currentCode"
      :language="currentLanguage"
      theme="dark"
    />
  </div>
</template>

<script setup lang="ts">
import { ref, watch } from "vue";
import CodeEditor from "./CodeEditor.vue";

// 每种语言的默认代码模板
const codeTemplates: Record<string, string> = {
  cpp: `#include <iostream>
using namespace std;

int main() {
    // 在此编写代码
    return 0;
}`,
  java: `import java.util.*;

public class Main {
    public static void main(String[] args) {
        // 在此编写代码
    }
}`,
  python: `# 在此编写代码
def main():
    pass

if __name__ == "__main__":
    main()`,
  go: `package main

import "fmt"

func main() {
    // 在此编写代码
}`,
  javascript: `// 在此编写代码
function main() {
}

main();`,
};

const currentLanguage = ref("cpp");
const currentCode = ref(codeTemplates.cpp);

// 切换语言时自动填充模板（仅当代码为空或匹配旧模板时）
const handleLanguageChange = (newLang: string) => {
  const oldTemplate = codeTemplates[currentLanguage.value];
  if (!currentCode.value || currentCode.value === oldTemplate) {
    currentCode.value = codeTemplates[newLang];
  }
};
</script>
```

## 4.4 Markdown 生态

### 编辑器（MdEditor）

基于 **ByteMD** 库，提供富文本 Markdown 编辑体验：

```vue
<!-- src/components/MdEditor.vue -->
<template>
  <Editor
    :value="modelValue"
    :plugins="plugins"
    @change="handleChange"
  />
</template>

<script setup lang="ts">
import { Editor } from "@bytemd/vue-next";
import gfm from "@bytemd/plugin-gfm";
import highlight from "@bytemd/plugin-highlight";

const plugins = [
  gfm(),       // GitHub Flavored Markdown（表格、任务列表等）
  highlight(), // 代码语法高亮
];

const props = defineProps<{ modelValue: string }>();
const emit = defineEmits(["update:modelValue"]);

const handleChange = (v: string) => {
  emit("update:modelValue", v);
};
</script>
```

### 预览器（MdPreview）

基于 **markdown-it** 库，支持代码高亮和数学公式：

```vue
<!-- src/components/MdPreview.vue -->
<template>
  <div class="markdown-body" v-html="htmlContent"></div>
</template>

<script setup lang="ts">
import { computed } from "vue";
import MarkdownIt from "markdown-it";
import highlight from "markdown-it-highlightjs";
import texmath from "markdown-it-texmath";
import katex from "katex";

const props = defineProps<{ value: string }>();

// 配置 markdown-it
const md = new MarkdownIt({
  html: false,  // 禁止原始 HTML（XSS 防护）
  linkify: true,
  typographer: true,
})
  .use(highlight)           // 代码语法高亮
  .use(texmath, { katex }); // LaTeX 数学公式

// 渲染 Markdown 为 HTML
const htmlContent = computed(() => {
  return md.render(props.value || "");
});
</script>
```

### 两个 Markdown 管线

| 用途 | 组件 | 库 | 特点 |
|------|------|-----|------|
| **编辑** | MdEditor | ByteMD | 工具栏、实时预览、GFM 支持 |
| **预览** | MdPreview | markdown-it | 代码高亮、LaTeX 公式、XSS 防护 |
